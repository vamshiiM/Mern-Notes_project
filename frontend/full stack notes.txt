import mongoose from "mongoose";

// create schema
// create model of that schema 

const noteSchema = new mongoose.Schema(
    {
        title: {
            type: String,
            required: true,
        },
        content: {
            type: String,
            required: true,
        }
    },
    { timestamps: true }
)

const Note = mongoose.model("Note", noteSchema)

export default Note;

rate limiting 

Rate limiting is a way to control how often someone can do something on a website or app like how many times they can refresh a page, make a request to an API, or try to Log in.

such as only 100 requests per user every 15 mins 

it prevents from too many requests and server overloading
